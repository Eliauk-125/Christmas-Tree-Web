    <canvas id="canvas"></canvas>
    <script>
        const canvas = document.getElementById('canvas');
        const ctx = canvas.getContext('2d');

        // 设置画布尺寸
        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        resizeCanvas();
        window.addEventListener('resize', resizeCanvas);

        // 粒子类
        class Particle {
            constructor(x, y, z, color) {
                this.x = x;
                this.y = y;
                this.z = z;
                this.color = color;
            }
        }

        // 生成圣诞树粒子
        const treeParticles = [];
        const treeHeight = 12;
        const treePointCount = 5000;
        for (let i = 0; i < treePointCount; i++) {
            const t = Math.random() * Math.PI * 2;
            const h = Math.random() * treeHeight;
            const r = 2 * h / treeHeight;
            const x = r * Math.cos(t);
            const y = h;
            const z = r * Math.sin(t);
            treeParticles.push(new Particle(x, y, z, `rgba(255, 192, 203, ${Math.random() * 0.8 + 0.2})`));
        }

        // 生成星星粒子
        const starParticles = [];
        const starPointCount = 1200;
        for (let i = 0; i < starPointCount; i++) {
            const x = (Math.random() - 0.5) * 30;
            const y = Math.random() * 15 + 8;
            const z = (Math.random() - 0.5) * 30;
            starParticles.push(new Particle(x, y, z, `rgba(255, 255, 255, ${Math.random() * 0.6 + 0.4})`));
        }

        // 生成地面粒子
        const groundParticles = [];
        const groundPointCount = 4000;
        for (let i = 0; i < groundPointCount; i++) {
            const x = (Math.random() - 0.5) * 10;
            const z = (Math.random() - 0.5) * 10;
            const y = 0;
            groundParticles.push(new Particle(x, y, z, `rgba(255, 192, 203, ${Math.random() * 0.5 + 0.3})`));
        }

        // 相机参数
        let camDist = 13;
        let camHeight = 6;
        let pitch = -0.25;

        // 动画循环
        function animate() {
            requestAnimationFrame(animate);
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // 合并所有粒子
            const allParticles = [...treeParticles, ...starParticles, ...groundParticles];

            // 绘制粒子
            allParticles.forEach(particle => {
                const dz = particle.z + camDist;
                if (dz < 0.1) return;
                const scale = 1 / dz;
                const x2d = (particle.x * scale) * canvas.width / 2 + canvas.width / 2;
                const y2d = (particle.y * scale - camHeight * scale) * canvas.height / 2 + canvas.height / 2;
                const size = Math.max(1, scale * 3);
                ctx.fillStyle = particle.color;
                ctx.beginPath();
                ctx.arc(x2d, y2d, size, 0, Math.PI * 2);
                ctx.fill();
            });
        }

        animate();
    </script>
